import matplotlib.pyplot as plt
import numpy as np

# Create x values
x = np.linspace(0, 10, 100)

# Define two functions
y1 = x**2           # Function 1
y2 = np.sqrt(x)     # Function 2

# Create the plot
plt.figure(figsize=(8,5))

# Plot first function as a solid blue line
plt.plot(x, y1, label='y = x^2', color='Pink', linestyle='-')

# Plot second function as red dashed line with markers
plt.plot(x, y2, label='y = sqrt(x)', color='Purple', linestyle='--', marker='o', markersize=4)

# Add labels, title, and legend
plt.xlabel('x')
plt.ylabel('y')
plt.title('Two Functions with Different Styles')
plt.legend()

# Display the plot
plt.show()

import matplotlib.pyplot as plt
import numpy as np

# Create x values
x = np.linspace(0, 10, 100)

# Define two functions
y1 = x**2
y2 = np.sqrt(x)

# Create subplots: 1 row, 2 columns
fig, axes = plt.subplots(1, 2, figsize=(12,5))

# Plot y = x^2 on the first subplot
axes[0].plot(x, y1, color='Pink', linestyle='-')
axes[0].set_title('y = x^2')
axes[0].set_xlabel('x')
axes[0].set_ylabel('y')

# Plot y = sqrt(x) on the second subplot
axes[1].plot(x, y2, color='Purple', linestyle='--', marker='o', markersize=4)
axes[1].set_title('y = sqrt(x)')
axes[1].set_xlabel('x')
axes[1].set_ylabel('y')

# Adjust spacing between plots
plt.tight_layout()

# Display the plots
plt.show()

import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Step 1: Load the Excel file
cancer_cases = pd.read_excel("Cancer_cases_deaths_2025.xlsx")

# Step 2: Clean the data (remove rows where Estimated New Cases or Estimated Deaths is "-")
cancer_cases = cancer_cases[(cancer_cases['Estimated New Cases 2025'] != "-") &
                            (cancer_cases['Estimated Deaths 2025'] != "-")]

# Step 3: Convert values to integers
cancer_cases['Estimated New Cases 2025'] = cancer_cases['Estimated New Cases 2025'].astype(int)
cancer_cases['Estimated Deaths 2025'] = cancer_cases['Estimated Deaths 2025'].astype(int)

# Step 4: Optional – take top 10 cancers
top_cancers = cancer_cases.head(10)

# Step 5: Prepare data for grouped bar plot
x = np.arange(len(top_cancers))  # the label locations
width = 0.35  # width of the bars

# Step 6: Create the plot
plt.figure(figsize=(12,6))
plt.bar(x - width/2, top_cancers['Estimated New Cases 2025'], width, label='New Cases', color='Pink')
plt.bar(x + width/2, top_cancers['Estimated Deaths 2025'], width, label='Deaths', color='Purple')

# Step 7: Add labels, title, and legend
plt.xticks(x, top_cancers['Common Type of Cancers'], rotation=45, ha='right')
plt.ylabel('Number of Cases')
plt.title('Top 10 Estimated New Cancer Cases vs Deaths in 2025')
plt.legend()
plt.tight_layout()
plt.show()
