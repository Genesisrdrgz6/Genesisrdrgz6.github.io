# Import necessary libraries
import matplotlib.pyplot as plt
import numpy as np

# Create x values
x = np.linspace(-10, 10, 400)  # 400 points from -10 to 10

# Compute f(x) = x^2
y = x**2

# Plot the function
plt.plot(x, y, label='f(x) = x^2', color='blue')

# Add labels and title
plt.xlabel('x')
plt.ylabel('f(x)')
plt.title('Graph of f(x) = x^2')
plt.legend()
plt.grid(True)

# Show the plot
plt.show()

# Create x values
x = np.linspace(-10, 10, 400)

# Compute functions
y1 = x**2       # f(x) = x^2

# Add labels and title

plt.ylabel('y')
plt.title('Graphs of f(x) = x^2 and g(x) = x^3')
plt.legend()
plt.grid(True)
